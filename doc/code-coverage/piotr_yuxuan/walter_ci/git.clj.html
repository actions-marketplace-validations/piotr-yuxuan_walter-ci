001  (ns piotr-yuxuan.walter-ci.git
002    (:require [piotr-yuxuan.walter-ci.files :refer [->tmp-file with-delete!]]
003              [babashka.process :as process]
004              [safely.core :refer [safely]])
005    (:import (java.io File)
006             (java.nio.file.attribute PosixFilePermissions PosixFilePermission)))
007  
008  (defn ^File askpass
009    "This is kind of a useless indirection. Perhaps ssh would be better?"
010    [secret-name]
011    (doto (->tmp-file "askpass" "sh" (PosixFilePermissions/asFileAttribute
012                                       #{PosixFilePermission/OWNER_READ
013                                         PosixFilePermission/OWNER_WRITE
014                                         PosixFilePermission/OWNER_EXECUTE}))
015      (spit (format "#!/bin/sh\necho \"${%s}\"\n" secret-name))))
016  
017  (defn clone
018    [^File working-directory {:keys [github-repository walter-github-password]}]
019    (with-delete! [askpass (askpass "GIT_PASSWORD")]
020      (assert (zero? (:exit @(process/process ["git"
021                                               ;; For this specific line, see https://github.com/actions/checkout/issues/162#issuecomment-590821598
022                                               "-c" "http.https://github.com/.extraheader="
023                                               "clone" "--depth" "1" (str "https://github.com/" github-repository)
024                                               "."]
025                                              {:out :inherit
026                                               :err :inherit
027                                               :dir (.getPath working-directory)
028                                               :env {"GIT_ASKPASS" (.getAbsolutePath askpass)
029                                                     "GIT_PASSWORD" walter-github-password}})))
030              "Failed to clone GitHub repository")))
031  
032  (defn stage-all
033    [^File working-directory _]
034    (assert (zero? (:exit @(process/process "git add --all"
035                                            {:out :inherit
036                                             :err :inherit
037                                             :dir (.getPath working-directory)})))
038            "Failed ot add updated files to index."))
039  
040  (defn need-commit?
041    [^File working-directory _]
042    (let [{:keys [out exit]} @(process/process "git diff --staged"
043                                               {:out :slurp
044                                                :err :inherit
045                                                :dir (.getPath working-directory)})]
046      (assert (zero? exit) "Failed to get staged changes in git.")
047      (boolean (seq out))))
048  
049  (defn commit
050    "Simple `git commit` and nothing else."
051    ;; FIXME https://git-scm.com/book/en/v2/Git-Tools-Signing-Your-Work
052    [^File working-directory {:keys [walter-git-email walter-author-name github-actor]} commit-message]
053    @(process/process ["git" "commit" "-m" commit-message]
054                      {:out :inherit
055                       :err :inherit
056                       :dir (.getPath working-directory)
057                       :env {"GIT_COMMITTER_NAME" github-actor
058                             "GIT_COMMITTER_EMAIL" walter-git-email
059                             "GIT_AUTHOR_NAME" walter-author-name
060                             "GIT_AUTHOR_EMAIL" walter-git-email}}))
061  
062  (defn push
063    "Simple `git push` and nothing else."
064    [^File working-directory {:keys [walter-github-password walter-git-email walter-author-name github-actor]}]
065    (with-delete! [askpass (askpass "GIT_PASSWORD")]
066      (safely
067        (process/check
068          @(process/process "git pull --rebase"
069                            {:out :inherit
070                             :err :inherit
071                             :dir (.getPath working-directory)
072                             :env {"GIT_COMMITTER_NAME" github-actor
073                                   "GIT_COMMITTER_EMAIL" walter-git-email
074                                   "GIT_AUTHOR_NAME" walter-author-name
075                                   "GIT_AUTHOR_EMAIL" walter-git-email
076                                   "GIT_PASSWORD" walter-github-password
077                                   "GIT_ASKPASS" (.getAbsolutePath askpass)}}))
078        (process/check
079          @(process/process ["git"
080                             ;; For this specific line, see https://github.com/actions/checkout/issues/162#issuecomment-590821598
081                             "-c" "http.https://github.com/.extraheader="
082                             "push"]
083                            {:out :inherit
084                             :err :inherit
085                             :dir (.getPath working-directory)
086                             :env {"GIT_PASSWORD" walter-github-password
087                                   "GIT_ASKPASS" (.getAbsolutePath askpass)}}))
088        :on-error
089        :max-retries 5)))
