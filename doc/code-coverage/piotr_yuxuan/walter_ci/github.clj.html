035  (def request-key-transformer
036    (-> {}
037        (assoc-in [:encoders :map :compile]
038          (fn [schema _]
039            (partial medley/map-keys
040                     (comp :request-key
041                           m/properties
042                           (partial mu/get schema)))))
043        (assoc-in [:decoders :map :compile]
044          (fn [schema _]
045            (let [request-keys (reduce (fn [acc [schema-key val-schema]]
046                                         (assoc acc
047                                           (->> val-schema
048                                                m/-children
049                                                first
050                                                m/properties
051                                                :request-key)
052                                           schema-key))
053                                       {}
054                                       (m/-entries schema))]
055              (fn [m]
056                (dissoc
057                  (medley.core/map-keys (fn [k] (get request-keys k ::not-found)) m)
058                  ::not-found)))))
059        mt/transformer))

061  (defn ensure-dynamic-classloader!
062    ;; https://github.com/riemann/riemann/pull/892/files#diff-691f7c44e67c3e00a8d2eda28e4b4422535c693efaf6779ce29aa537fef26ceeR109-R115
063    [^Thread current-thread]
064    (let [cl (.getContextClassLoader current-thread)]
065      (when-not (instance? DynamicClassLoader cl)
066        (.setContextClassLoader current-thread (DynamicClassLoader. cl)))))

068  (defn expected-settings
069    [{:keys [github-workspace]}]
070    (let [project (try (ensure-dynamic-classloader! (Thread/currentThread))
071                       (leiningen/read (.getAbsolutePath (io/file github-workspace "project.clj")) [:github])
072                       (catch Throwable th
073                         (println th)
074                         (println (ex-message th))
075                         (println (ex-data th))
076                         (println (ex-cause th))))]
077      (m/encode Defaults (-> project
078                             (assoc :github/homepage (or (:github/homepage project) (:url project) (:scm (:url project))))
079                             (assoc :github/description (or (:github/description project) (:description project))))
080                (mt/transformer
081                  mt/default-value-transformer
082                  mt/strip-extra-keys-transformer))))

084  (defn apply-settings
085    [{:keys [github-api-url github-actor github-repository walter-github-password]} {:github/keys [topics] :as settings}]
086    ;; Setting topics through the API is still in preview.
087    (safely
088      (http/request {:request-method :put
089                     :url (str/join "/" [github-api-url "repos" github-repository "topics"])
090                     :body (json/write-value-as-string {:names topics})
091                     :basic-auth [github-actor walter-github-password]
092                     :headers {"Content-Type" "application/json"
093                               "Accept" "application/vnd.github.mercy-preview+json"}})
094      :on-error
095      :max-retries 5)
096    (let [json-settings (m/encode Defaults (dissoc settings :github/topics)
097                                  (mt/transformer
098                                    mt/strip-extra-keys-transformer
099                                    request-key-transformer))]
100      (safely
101        (http/request {:request-method :patch
102                       :url (str/join "/" [github-api-url "repos" github-repository])
103                       :body (json/write-value-as-string json-settings)
104                       :basic-auth [github-actor walter-github-password]
105                       :headers {"Content-Type" "application/json"
106                                 "Accept" "application/vnd.github.v3+json"}})
107        :on-error
108        :max-retries 5)))
109  
110  (defn conform-repository
111    [config]
112    (apply-settings config (expected-settings config)))
