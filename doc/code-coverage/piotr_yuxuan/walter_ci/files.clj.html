011  (defn ^File ->file
012    "Do-what-I-mean syntactic sugar that return the canonical file for its arguments. Return nil if not possible."
013    ([] (->file "."))
014    ([x & r]
015     (letfn [(join-seq [c]
016               (->> (flatten c) ; Perhaps a legit use case of flatten?
017                    (map path-elem)
018                    (remove nil?)
019                    (str/join (File/separator))))
020             (path-elem [x]
021               (as-> x $
022                 (if (instance? Named $) (name $) $)
023                 (if (instance? Path $) (.toFile ^Path $) $)
024                 (if (instance? URI $) (.getPath ^URI $) $)
025                 (if (instance? URL $) (.getFile ^URL $) $)))]
026       ;; Try to bring anything in line with the happy path.
027       (as-> x $
028         (if (seq r) (cons x r) $)
029         (if (sequential? $) (join-seq $) $)
030         (path-elem $)
031         (if (string? $) (io/file $) $)
032         (when (instance? File $)
033           (.getCanonicalFile ^File $))))))

035  (defn ^File ->dir
036    "Do-what-I-mean syntactic sugar that return the canonical directory file for its arguments. Return nil if not possible."
037    ([& args]
038     (when-let [^File f (apply ->file args)]
039       (when (.isDirectory f)
040         f))))

042  (defn ^File ->tmp-dir
043    ([] (->tmp-dir nil nil))
044    ([prefix] (->tmp-dir prefix nil))
045    ([prefix file-attributes]
046     (-> prefix
047         ^Path (Files/createTempDirectory (into-array FileAttribute file-attributes))
048         .toFile
049         .getCanonicalFile)))

051  (defn ^File ->tmp-file
052    ([] (->tmp-file nil nil nil))
053    ([suffix] (->tmp-file nil suffix nil))
054    ([prefix suffix] (->tmp-file prefix suffix nil))
055    ([prefix suffix & file-attributes]
056     (-> prefix
057         ^Path (Files/createTempFile suffix (into-array FileAttribute file-attributes))
058         .toFile
059         .getCanonicalFile)))

061  (defn delete!
062    "Idempotent procedure that ensures that the argument `file` no longer exists at the end."
063    [^File file & [silently]]
064    (when (.isDirectory file)
065      (->> (.listFiles file)
066           (pmap #(delete! % silently))
067           doall))
068    (io/delete-file file silently)
069    nil)

071  (defn copy!
072    [^File source ^File target
073     & {:keys [callback]
074        :or {callback (fn [_source _target] nil)}
075        :as opts}]
076    (if (.isDirectory source)
077      (->> (.listFiles source)
078           (pmap (fn [nested-source]
079                   (let [nested-target (io/file target (.getName ^File nested-source))]
080                     (copy! nested-source nested-target)
081                     (callback nested-source nested-target))))
082           doall)
083      (apply io/copy source target opts))
084    nil)

086  (defmacro with-delete!
087    "bindings => [name init ...]
088  
089    Evaluates body in a try expression with names bound to the values
090    of the inits, and a `finally` clause that deletes each file in reverse order."
091    {:added "1.0"}
092    [bindings & body]
093    (#'clojure.core/assert-args
094      (vector? bindings) "a vector for its binding"
095      (even? (count bindings)) "an even number of forms in binding vector")
096    (cond
097      (zero? (count bindings)) `(do :no-kibit ~@body)
098      (symbol? (bindings 0)) `(let ~(subvec bindings 0 2)
099                                (try
100                                  (with-delete! ~(subvec bindings 2) ~@body)
101                                  (finally
102                                    (~delete! ~(bindings 0)))))
103      :else (throw (IllegalArgumentException.
104                     "with-delete only allows Symbols in bindings"))))
