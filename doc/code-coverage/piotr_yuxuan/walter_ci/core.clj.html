015  (defn update-workflow
016    [options ^File workflow-file]
017    (with-delete! [working-directory (->tmp-dir "update-workflow")]
018      (git/clone working-directory options)
019      (io/copy workflow-file (doto (->file working-directory ".github" "workflows" (.getName workflow-file))
020                               (io/make-parents)))
021      (git/stage-all working-directory options)
022      (when (git/need-commit? working-directory options)
023        (git/commit working-directory options (format "Update %s" (.getName workflow-file)))
024        (git/push working-directory options))))

026  (defn replicate-walter-ci
027    [{:keys [github-action-path managed-repositories] :as config}]
028    (doseq [github-repository managed-repositories]
029      (let [config+github-repository (assoc config :github-repository github-repository)]
030        (doseq [secret-name [:walter-author-name
031                             :walter-github-password
032                             :walter-git-email]]
033          (secret/upsert-value config+github-repository
034                               (csk/->SCREAMING_SNAKE_CASE_STRING secret-name)
035                               (get config secret-name)))
036        (update-workflow config+github-repository (->file github-action-path "resources" "workflows" "walter-ci.yml")))))

054    [{:keys [^File github-workspace] :as config}]
055    (let [^File github-workspace (io/file ".")
056          ^File txt-report (doto (io/file "./doc/Known vulnerabilities.txt") io/make-parents)]
057      (delete! (io/file "./doc/Known vulnerabilities.md") :quiet)
058      (delete! (io/file "./doc/Known vulnerabilities.csv") :quiet)
059      (delete! (io/file "./doc/KNOWN_VULNERABILITIES.md") :quiet)
060      (with-open [txt-report-writer (io/writer txt-report)]
061        @(process/process ["clojure" "-Sdeps" (pr-str {:aliases {:nvd {:extra-deps {'nvd-clojure/nvd-clojure {:mvn/version "LATEST"}}}}})
062                           "-M:nvd"
063                           "-m" "nvd.task.check"]
064                          {:out txt-report-writer
065                           :err :inherit
066                           :dir (.getPath github-workspace)}))
067      ;; Remove ANSI colour codes.
068      (spit txt-report (str/replace (slurp txt-report) #"\x1b\[[0-9;]*m" "")))
069    (git/stage-all github-workspace config)
070    (when (git/need-commit? github-workspace config)
071      (assert (zero? (:exit (git/commit github-workspace config "Report vulnerabilities")))
072              "Failed to commit vulnerability report.")
073      (git/push github-workspace config)))

075  (defn clojure-git-ignore
076    [{:keys [github-action-path github-workspace] :as options}]
077    (let [required-entries (set (line-seq (io/reader (->file github-action-path "resources" ".template-gitignore"))))
078          current-entries (set (line-seq (io/reader (->file github-workspace ".gitignore"))))
079          missing-entries (sort (set/difference required-entries current-entries))]
080      (spit (->file github-workspace ".gitignore")
081            (str/join (System/lineSeparator) missing-entries)
082            :append true)
083      (git/stage-all github-workspace options)
084      (when (git/need-commit? github-workspace options)
085        (git/commit github-workspace options (format "Update .gitignore"))
086        (git/push github-workspace options))))

144  (defn package-deploy-artifacts
145    [{:keys [github-workspace]}]
146    (let [leiningen-project (-> (io/file github-workspace "project.clj")
147                                (.getAbsolutePath)
148                                (leiningen/read [:deploy]))
149          deploy-repositories (->> leiningen-project
150                                   :deploy-repositories
151                                   (map first)
152                                   seq)]
153      (if deploy-repositories
154        (println "Deploying to repositories:" deploy-repositories)
155        (println "No deploy repositories found, not deploying."))
156      (doseq [deploy-repository deploy-repositories]
157        (println :dry-run ["lein" "deploy" deploy-repository])
158        #_(let [{:keys [exit]} @(process/process ["lein" "deploy" deploy-repository]
159                                                 {:out :inherit
160                                                  :err :inherit
161                                                  :dir (.getPath github-workspace)})]
162            (when-not (zero? exit)
163              (println "Deployment failed to" deploy-repository))))))
